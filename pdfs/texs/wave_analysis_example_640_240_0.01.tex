\section*{Анализ затухания волн: L=640, DW=240, R=0.01}

\subsection*{Параметры моделирования}

\begin{itemize}
    \item Длина образца: $L = 640$ м
    \item Ширина образца: $DW = 240$ м
    \item Коэффициент затухания: $R = 0.01$
    \item Модуль Юнга: $E$, Коэффициент Пуассона: $\nu$, Плотность: $\rho$
\end{itemize}

\subsection*{Расчётные скорости волн}

\begin{itemize}
    \item Аналитическая скорость продольной волны: $v_p = 376.43$ м/с
    \item Аналитическая скорость поперечной волны: $v_s = 201.21$ м/с
    \item Аналитическая скорость Рэлеевской волны: $v_r = 186.35$ м/с
    \item Временной шаг: $\Delta t = 3.41e-04$ с
    \item Экспериментальная скорость поперечной волны: $v_s^{exp} = 250.00$ м/с
    \item Экспериментальная скорость отраженной продольной волны: $v_p^{ref} = 192.25$ м/с
    \item Величина ошибки: $\varepsilon = 9.41e-08$
\end{itemize}

\subsection*{Анализ прихода продольной волны}

\begin{itemize}
    \item Аналитическое время прихода продольной волны: $t_{analytical} = 1.70$ с
    \item Экспериментальный индекс прихода продольной волны: 5098
    \item Экспериментальное время прихода продольной волны: $t_{wave} = 1.74$ с
    \item Скорость пришедшей продольной волны: $v_p^{exp} = 368.37$ м/с
    \item Длина продольной волны в индексах: 1338
    \item Длина продольной волны в метрах: $\lambda_p = 167.93$ м
\end{itemize}

\subsection*{Анализ пересечения волн}

\begin{itemize}
    \item Экспериментальное время пересечения волн: $t_{intersection} = 2.20$ с
    \item Экспериментальный индекс пересечения волн: 6463
    \item Экспериментальное расстояние пересечения волн: $x_{intersection} = 550.60$ м
    \item Номер приемника, ближайшего к пересечению: 28
    \item Индекс прихода падающей волны через ресивер пересечения: 4461
\end{itemize}
